# Plan: MILP Workout API — Comprehensive Fix

## Goal
Fix all critical issues in `POST /workout-milp/generate` endpoint to produce correct, realistic workout plans that:
- Cover mandatory muscles (including group muscles like `back` → `lats`/`upper_back`/`traps`)
- Fill the time budget realistically
- Respect equipment constraints
- Run efficiently (<1s latency)
- Provide transparency via response flags

## Problem Summary

| # | Issue | Severity | Root Cause |
|---|-------|----------|------------|
| 1 | `back` never covered | 🔴 Critical | No mapping from `back` → `lats`/`upper_back`/`traps` |
| 2 | `timeCostSec` ignores sets | 🔴 Critical | Seed assigns per-exercise, not per-set; no ×sets in LP |
| 3 | N+1 queries (7465 SQL) | 🔴 Critical | 5 sub-queries per exercise in `toExercise()` |
| 4 | `fatigueByMuscle`/`usedExercises` empty | 🔴 Critical | Controller passes `{}` and `[]` hardcoded |
| 5 | Response missing flags | 🟡 Moderate | DTO doesn't expose `usedFallback`/`partialCoverage` |
| 6 | Equipment filter bypassed on `[]` | 🟡 Moderate | `if (availableEquipment.length > 0)` skips filter |
| 7 | `'body weight'` case-sensitive | 🟡 Moderate | Slug comparison without normalization |
| 8-11 | Minor issues | 🟢 Low | See analysis section |

```typescript
private checkCoverageFeasibility(candidates: Exercise[], input: WorkoutMILPInput): CoverageCheck {
  const mandatory = input.mandatoryMuscles ?? [];
  const coverable = new Set<string>();

  if (mandatory.length === 0) {
    return { mandatory, coverable };
  }

  // Expand group muscles (back → [lats, upper_back, traps])
  const expandedMandatory = new Set<string>();
  for (const m of mandatory) {
    const expanded = this.expandMuscleGroup(m);
    expanded.forEach(e => expandedMandatory.add(e));
  }

  for (const ex of candidates) {
    const meta = ex.metadata ?? {};
    for (const mw of meta.primaryMuscleWeights ?? []) {
      const slug = this.normalizeSlug(mw.slug);
      if (expandedMandatory.has(slug)) coverable.add(slug);
    }
    for (const mw of meta.secondaryMuscleWeights ?? []) {
      const slug = this.normalizeSlug(mw.slug);
      if (expandedMandatory.has(slug)) coverable.add(slug);
    }
  }

  return { mandatory, coverable, expandedMandatory };
}
```

```typescript
async findForMILP(page: number, limit: number): Promise<PaginatedResult<ExerciseMILPData>> {
  const offset = (page - 1) * limit;
  
  const countRow = await this.db.queryOne<{ total: string }>(
    `SELECT COUNT(*)::text AS total FROM exercises`,
    []
  );
  const total = parseInt(countRow?.total ?? '0', 10);

  // Single query with JOINs for equipment and contraindications
  const rows = await this.db.query<{
    slug: string;
    exercise_type: string | null;
    movement_pattern: string | null;
    metadata: ExerciseMetadata | null;
    equipments: { slug: string }[];
    contraindications: { slug: string; severity: string }[];
  }>(
    `SELECT 
       e.slug,
       e.exercise_type,
       e.movement_pattern,
       e.metadata,
       COALESCE(
         (SELECT json_agg(json_build_object('slug', eq.slug))
          FROM exercise_equipments ee
          JOIN equipments eq ON eq.id = ee.equipment_id
          WHERE ee.exercise_id = e.id),
         '[]'::json
       ) as equipments,
       COALESCE(
         (SELECT json_agg(json_build_object('slug', c.slug, 'severity', ec.severity))
          FROM exercise_contraindications ec
          JOIN contraindications c ON c.id = ec.contraindication_id
          WHERE ec.exercise_id = e.id),
         '[]'::json
       ) as contraindications
     FROM exercises e
     ORDER BY e.slug
     LIMIT $1 OFFSET $2`,
    [limit, offset]
  );

  const data: ExerciseMILPData[] = rows.map(r => ({
    slug: r.slug,
    exerciseType: r.exercise_type,
    movementPattern: r.movement_pattern,
    metadata: r.metadata,
    equipments: r.equipments.map(e => e.slug),
    contraindications: r.contraindications,
  }));

  return { data, total };
}
```

```typescript
private async computeFatigueAndHistory(
  userId: string,
  daysBack: number = 14
): Promise<{ fatigueByMuscle: Record<string, number>; usedExercises: string[] }> {
  const sessions = await this.sessionsRepository.findRecentCompletedByUserId(userId, daysBack);
  
  const fatigueByMuscle: Record<string, number> = {};
  const exerciseCounts: Record<string, number> = {};
  
  for (const session of sessions) {
    for (const ex of session.exercises ?? []) {
      // Count exercise usage
      exerciseCounts[ex.exerciseSlug] = (exerciseCounts[ex.exerciseSlug] ?? 0) + 1;
      
      // Accumulate fatigue per muscle from exercise metadata
      const exercise = await this.exercisesRepository.findBySlug(ex.exerciseSlug);
      if (exercise?.metadata) {
        for (const mw of exercise.metadata.primaryMuscleWeights ?? []) {
          const slug = this.normalizeSlug(mw.slug);
          const load = (exercise.metadata.fatigueCost ?? 5) * mw.weight * ex.sets;
          fatigueByMuscle[slug] = (fatigueByMuscle[slug] ?? 0) + load;
        }
      }
    }
  }
  
  // Decay fatigue by recency (most recent = full, oldest = 30%)
  const now = Date.now();
  const maxAge = daysBack * 24 * 60 * 60 * 1000;
  for (const session of sessions) {
    const age = now - new Date(session.metadata?.sessionDurationMin ?? 0).getTime();
    const decay = Math.max(0.3, 1 - (age / maxAge) * 0.7);
    // Apply decay (simplified - real implementation would track per-muscle timestamps)
  }
  
  const usedExercises = Object.keys(exerciseCounts);
  
  return { fatigueByMuscle, usedExercises };
}
```

### Phase 5: Equipment Filter Fix (Issues #6, #7)

```typescript
private filterCandidates(
  exercises: ExerciseMILPData[],
  input: WorkoutMILPInput,
): ExerciseMILPData[] {
  return exercises.filter((ex) => {
    // Exclude forbidden contraindications
    if (input.userContraindications?.length && ex.contraindications?.length) {
      for (const c of ex.contraindications) {
        if (
          input.userContraindications.includes(c.slug) &&
          c.severity === 'forbidden'
        ) {
          return false;
        }
      }
    }

    // Equipment filtering
    const equipments = ex.equipments ?? [];
    const availableEquipment = input.availableEquipment ?? [];
    
    // Normalize equipment slugs for comparison
    const normalizeEquipment = (slug: string) => slug.toLowerCase().replace(/[-_\s]/g, '');
    const isBodyweight = (slug: string) => {
      const normalized = normalizeEquipment(slug);
      return ['bodyweight', 'bodyweight', 'bodyweight', 'none'].includes(normalized);
    };
    
    // If no available equipment specified → only bodyweight allowed
    if (availableEquipment.length === 0) {
      const hasOnlyBodyweight = equipments.length === 0 || equipments.every(isBodyweight);
      if (!hasOnlyBodyweight) return false;
    } else {
      // Equipment specified → must have at least one matching
      const hasEquipment = equipments.some(eq => 
        availableEquipment.some(avail => 
          normalizeEquipment(eq) === normalizeEquipment(avail)
        )
      );
      const needsEquipment = equipments.length > 0 && !equipments.some(isBodyweight);
      
      if (needsEquipment && !hasEquipment) return false;
    }

    return true;
  });
}
```
